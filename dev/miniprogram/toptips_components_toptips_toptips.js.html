let topTipsData = {
    topTipsCtx: null,
    timeId: 0
};

/**
 * Toptips 用于顶部下滑通知。
 * 
 * 请下载 [代码片段](wechatide://minicode/k7itLEm26MZ3)
 * @function Toptips 组件
 * @param {Object} options 传入设置
 * @example
 * // index.json
 * "usingComponents": {
 *  "toptips": "../components/toptips/index"
 * }
 * // index.wxml
 * <button bindtap="showToptips"></button>
 * <toptips id="zan-toptips"/>
 * // index.js
 * const Toptips = require('../components/toptips/toptips');
 * Page({
 *  data: {},
 *  showToptips() {
 *    Toptips({
 *      selector: '#zan-toptips',
 *      content: '测试内容',
 *      duration: 1000
 *    })
 *  }
 */
function Toptips(options = {}) {
    const ctx = getPageCtx();

    options = parseParam(options);

    // 获取目标操作组件实例
    const topTipsCtx = ctx.selectComponent(options.selector);

    if (!topTipsCtx) {
        console.error('无法找到对应的 toptips 组件，请于页面中注册并在 wxml 中声明 toptips 自定义组件');
        return;
    }

    // 在下一个 tips 启动前清除上一个
    if (topTipsData.timeId) {
        Toptips.clear();
    }

    topTipsCtx.show({
        ...options,
        isShow: true
    });

    const timeId = setTimeout(() => {
        Toptips.clear();
    }, options.duration);

    topTipsData = {
        topTipsCtx,
        timeId
    };
}

Toptips.clear = function() {
    clearTimeout(topTipsData.timeId);

    topTipsData.topTipsCtx && topTipsData.topTipsCtx.hide();

    topTipsData = {
        topTipsCtx: null,
        timeId: 0
    };
};

// 获取页面上下文
function getPageCtx() {
    const pages = getCurrentPages();
    return pages[pages.length - 1];
}

// 参数处理
function parseParam({selector, content, duration}) {
    let params = {};
    params['selector'] = toString.call(selector) === '[object String]' ? selector : null;
    params['content'] = toString.call(content) === '[object String]' ? content : '请填入内容';
    params['duration'] = toString.call(duration) === '[object Number]' ? duration : 3000;
    return params;
}

module.exports = Toptips;

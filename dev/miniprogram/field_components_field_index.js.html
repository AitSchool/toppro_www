/**
 * Field 组件是一个常用的表单组件。
 * 
 * 请下载 [代码片段](wechatide://minicode/X9stUEmi6lZO)
 * @function Field 组件
 * @param {String} title 表单字段名
 * @param {String} type textarea 或者 input 类型，默认值 input
 * @param {Boolean} disabled 是否禁用
 * @param {String} inputType input 类型：text,number,idcard,digit，默认值 text
 * @param {String} placeholder 占位符
 * @param {Boolean} focus 获取焦点
 * @param {String} mode 表单样式，默认值 normal
 * @param {Boolean} right 表单内容位置，默认左边
 * @param {Boolean} error true 时为红色
 * @param {Number} maxlength 表单内容字数上限，默认值 140
 */
function Properties() {
    return {
        title: String,
        type: {
            type: String,
            value: 'input'
        },
        disabled: Boolean,
        inputType: {
            type: String,
            value: 'text'
        },
        placeholder: String,
        focus: Boolean,
        mode: {
            type: String,
            value: 'normal'
        },
        right: Boolean,
        error: Boolean,
        maxlength: {
            type: Number,
            value: 140
        }
    };
}

    methods: {
        handleFieldChange(event) {
            const { detail = {} } = event;
            const { value = '' } = detail;
            this.setData({ value });

            this.triggerEvent('change', event);
        },

        handleFieldFocus(event) {
            this.triggerEvent('focus', event);
        },

        handleFieldBlur(event) {
            this.triggerEvent('blur', event);
        }
    }
});

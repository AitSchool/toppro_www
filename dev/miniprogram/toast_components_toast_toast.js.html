let timeoutData = {
    timeoutId: 0,
    toastCtx: null
};

/**
 * Toast 组件用于页面弹出通知，或者带 loading 状态的通知。
 * 
 * 请下载 [代码片段](wechatide://minicode/O0l30Em46QZU)
 * @function Toast 组件
 * @param {Object} options 传入设置
 * @example
 * // component.html
 * <button bindtap="openMessageToast"></button>
 * <toast id="selector"></toast>
 * // index.json
 * "usingComponents": {
 *  "toast": "../components/toast/index"
 * }
 * // index.js
 * const Toast = require('../components/toast/toast');
 * Page({
 *  data: {},
 *  openMessageToast() {
 *    Toast({
 *      selector: '#selector',
 *      message: '弹出内容',
 *      type: 'loading'
 *    })
 *  }
 * });
 */
function Toast(options) {

    // 参数格式化处理
    // 如果是文字，默认为 message
    options = parseParam(options);

    const ctx = getPageCtx();
    const toastCtx = ctx.selectComponent(options.selector);

    if (!toastCtx) {
        console.error('无法找到对应的toast组件，请于页面中注册并在 wxml 中声明 toast 自定义组件');
        return;
    }

    // 关闭已经存在的 Toast
    if (timeoutData.timeoutId) {
        Toast.clear();
    }

    toastCtx.show({
        ...options,
        show: true
    });

    const timeoutId = setTimeout(() => {
        toastCtx.clear();
    }, options.timeout || 3000);

    timeoutData = {
        timeoutId,
        toastCtx
    };
}

// 清理所有 toast
Toast.clear = function() {
    clearTimeout(timeoutData.timeoutId);

    timeoutData.toastCtx && timeoutData.toastCtx.clear();

    // 关闭 Toast 重置数据
    timeoutData = {
        timeoutId: 0,
        toastCtx: null
    };
};

// 获取页面上下文
function getPageCtx() {
    const pages = getCurrentPages();
    return pages[pages.length - 1];
}

// 参数处理
function parseParam({selector, message, type}) {
    let params = {};
    params['selector'] = toString.call(selector) === '[object String]' ? selector : null;
    params['message'] = toString.call(message) === '[object String]' ? message : '请填入内容';
    params['type'] = toString.call(type) === '[object String]' ? type : '';
    return params;
}

module.exports = Toast;
